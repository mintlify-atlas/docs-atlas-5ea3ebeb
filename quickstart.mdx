**Get a working message queue up and running in minutes.** This guide walks you through creating a publisher and consumer with Redis (the default broker).

<Steps>
  <Step title="Add Broccoli to Cargo.toml">
    Add the dependency with the Redis feature (default):

    ```toml
    [dependencies]
    broccoli_queue = "0.4.6"
    tokio = { version = "1.42", features = ["full"] }
    serde = { version = "1.0", features = ["derive"] }
    serde_json = "1.0"
    chrono = { version = "0.4", features = ["serde"] }
    ```

    Or with RabbitMQ instead:
    ```toml
    broccoli_queue = { version = "0.4.6", features = ["rabbitmq"] }
    ```

    Or SurrealDB:
    ```toml
    broccoli_queue = { version = "0.4.6", features = ["surrealdb"] }
    ```
  </Step>

  <Step title="Start Your Message Broker">
    **For Redis** (default):
    ```bash
    docker run -p 6380:6379 redis:7-alpine
    ```

    **For RabbitMQ**:
    ```bash
    docker run -p 5672:5672 -p 15672:15672 rabbitmq:3.12-management-alpine
    ```

    **For SurrealDB**:
    ```bash
    docker run -p 8000:8000 surrealdb/surrealdb:latest start --log info -- file://data.db
    ```
  </Step>

  <Step title="Define Your Message Type">
    Create a struct that represents your job payload. It must derive `Serialize` and `Deserialize`:

    ```rust
    use serde::{Deserialize, Serialize};

    #[derive(Debug, Clone, Serialize, Deserialize)]
    struct JobPayload {
        id: String,
        task_name: String,
        parameters: Parameters,
        created_at: chrono::DateTime<chrono::Utc>,
    }

    #[derive(Debug, Clone, Serialize, Deserialize)]
    struct Parameters {
        input_data: String,
        priority: u8,
        timeout_seconds: u32,
    }
    ```
  </Step>

  <Step title="Initialize the Queue">
    Build a `BroccoliQueue` instance with your broker URL:

    ```rust
    use broccoli_queue::queue::BroccoliQueue;

    let queue = BroccoliQueue::builder("redis://localhost:6380")
        .failed_message_retry_strategy(Default::default())
        .pool_connections(5)
        .build()
        .await?;
    ```

    For RabbitMQ, use `amqp://localhost:5672`.  
    For SurrealDB, use `ws://localhost:8000`.
  </Step>

  <Step title="Publish Messages">
    Send jobs to the queue for processing. You can publish individually or in batch:

    ```rust
    let jobs = vec![
        JobPayload {
            id: "job-1".to_string(),
            task_name: "process_data".to_string(),
            parameters: Parameters {
                input_data: "data-1".to_string(),
                priority: 1,
                timeout_seconds: 300,
            },
            created_at: chrono::Utc::now(),
        },
        JobPayload {
            id: "job-2".to_string(),
            task_name: "generate_report".to_string(),
            parameters: Parameters {
                input_data: "data-2".to_string(),
                priority: 2,
                timeout_seconds: 600,
            },
            created_at: chrono::Utc::now(),
        },
    ];

    let published = queue
        .publish_batch("jobs", None, jobs, None)
        .await?;

    println!("Published {} jobs", published.len());
    ```
  </Step>

  <Step title="Consume and Process Messages">
    Start workers to process messages from the queue:

    ```rust
    async fn process_job(job: JobPayload) -> Result<(), broccoli_queue::error::BroccoliError> {
        println!("Processing job: {} ({})", job.id, job.task_name);
        
        // Simulate work
        tokio::time::sleep(std::time::Duration::from_secs(1)).await;
        
        println!("Job {} completed", job.id);
        Ok(())
    }

    queue
        .process_messages("jobs", Some(4), None, |msg| async move {
            process_job(msg.payload).await
        })
        .await?;
    ```
  </Step>
</Steps>

## Complete Publisher Example

Here's a fully working publisher based on the official examples:

<CodeGroup>
```rust src/bin/publisher.rs
use broccoli_queue::{
    error::BroccoliError,
    queue::{BroccoliQueue, PublishOptions},
};
use serde::{Deserialize, Serialize};
use time::Duration;

#[derive(Debug, Clone, Serialize, Deserialize)]
struct JobPayload {
    id: String,
    task_name: String,
    parameters: Parameters,
    created_at: chrono::DateTime<chrono::Utc>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
struct Parameters {
    input_data: String,
    priority: u8,
    timeout_seconds: u32,
}

#[tokio::main]
async fn main() -> Result<(), BroccoliError> {
    let queue = BroccoliQueue::builder("redis://localhost:6380")
        .failed_message_retry_strategy(Default::default())
        .pool_connections(5)
        .build()
        .await?;

    // Create some example jobs
    let scheduled_jobs = vec![
        JobPayload {
            id: "job-1".to_string(),
            task_name: "process_data".to_string(),
            parameters: Parameters {
                input_data: "data-1".to_string(),
                priority: 1,
                timeout_seconds: 300,
            },
            created_at: chrono::Utc::now(),
        },
        JobPayload {
            id: "job-2".to_string(),
            task_name: "generate_report".to_string(),
            parameters: Parameters {
                input_data: "data-2".to_string(),
                priority: 2,
                timeout_seconds: 600,
            },
            created_at: chrono::Utc::now(),
        },
    ];

    // Publish jobs with a 10-second delay
    println!("Publishing delayed jobs...");
    let scheduled_jobs = queue
        .publish_batch(
            "jobs",
            None,
            scheduled_jobs,
            Some(PublishOptions {
                delay: Some(Duration::seconds(10)),
                scheduled_at: None,
                ttl: None,
                priority: None,
            }),
        )
        .await?;

    println!(
        "Published scheduled jobs: {:?}",
        scheduled_jobs.iter().map(|j| j.task_id).collect::<Vec<_>>()
    );

    let immediate_jobs = vec![
        JobPayload {
            id: "job-3".to_string(),
            task_name: "process_data".to_string(),
            parameters: Parameters {
                input_data: "data-3".to_string(),
                priority: 1,
                timeout_seconds: 300,
            },
            created_at: chrono::Utc::now(),
        },
        JobPayload {
            id: "job-4".to_string(),
            task_name: "generate_report".to_string(),
            parameters: Parameters {
                input_data: "data-4".to_string(),
                priority: 2,
                timeout_seconds: 600,
            },
            created_at: chrono::Utc::now(),
        },
    ];

    // Publish jobs immediately
    println!("Publishing immediate jobs...");
    let immediate_jobs = queue
        .publish_batch("jobs", None, immediate_jobs, None)
        .await?;

    println!(
        "Published immediate jobs: {:?}",
        immediate_jobs.iter().map(|j| j.task_id).collect::<Vec<_>>()
    );

    Ok(())
}
```
</CodeGroup>

## Complete Consumer Example

Here's a fully working consumer that processes messages with success and error handlers:

<CodeGroup>
```rust src/bin/consumer.rs
use broccoli_queue::{error::BroccoliError, queue::BroccoliQueue};
use serde::{Deserialize, Serialize};
use std::{error::Error, time::Duration};

#[derive(Debug, Clone, Serialize, Deserialize)]
struct JobPayload {
    id: String,
    task_name: String,
    parameters: Parameters,
    created_at: chrono::DateTime<chrono::Utc>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
struct Parameters {
    input_data: String,
    priority: u8,
    timeout_seconds: u32,
}

async fn process_job(job: JobPayload) -> Result<(), BroccoliError> {
    println!("Processing job: {} ({})", job.id, job.task_name);

    // Simulate some work
    tokio::time::sleep(Duration::from_secs(1)).await;

    println!("Completed job: {}", job.id);
    Ok(())
}

async fn success_handler(msg: JobPayload) -> Result<(), BroccoliError> {
    println!("Successfully processed message: {}", msg.id);
    Ok(())
}

async fn error_handler(msg: JobPayload, err: BroccoliError) -> Result<(), BroccoliError> {
    eprintln!("Failed to process message: {}. Error: {}", msg.id, err);
    Ok(())
}

#[tokio::main]
async fn main() -> Result<(), Box<dyn Error>> {
    env_logger::init();

    // Initialize the queue
    let queue = BroccoliQueue::builder("redis://localhost:6380")
        .pool_connections(5)
        .failed_message_retry_strategy(Default::default())
        .build()
        .await?;

    // Process messages with custom success and error handlers
    queue
        .process_messages_with_handlers(
            "jobs",
            Some(4),
            None,
            |msg| async move { process_job(msg.payload).await },
            |msg, _result| async { success_handler(msg.payload).await },
            |msg, err| async { error_handler(msg.payload, err).await },
        )
        .await?;

    println!("Consumer shutdown complete");
    Ok(())
}
```
</CodeGroup>

## Running the Examples

In separate terminals:

**Terminal 1: Run the publisher**
```bash
cargo run --bin publisher
```

**Terminal 2: Run the consumer**
```bash
cargo run --bin consumer
```

You'll see the publisher sending jobs and the consumer processing them in real-time.
